        function renderChatMessages() {
            const container = document.getElementById('chatMessagesList');
            if (!container) return;
            
            if (chatMessagesList.length === 0) {
                container.innerHTML = '<div style="text-align: center; color: #666; padding: 20px;">No messages yet. Be the first to chat!</div>';
                return;
            }
            
            const isAdmin = (currentUser && currentUsername === 'admin') || (chatUsername === 'Admin');
            
            container.innerHTML = chatMessagesList.map(msg => {
                const isOwnMessage = (msg.username === chatUsername);
                const isBlockedUser = blockedUsers.includes(msg.username);
                
                if (isBlockedUser && !isAdmin) {
                    return null;
                }
                
                return `
                    <div class="chat-message ${isOwnMessage ? 'own' : 'other'}">
                        <div class="message-sender">
                            ${escapeHtml(msg.username)}
                            ${msg.username === 'Admin' ? '<span class="admin-badge">Admin</span>' : ''}

                        <div class="message-time">${formatChatTime(msg.created_at)}</div>
                    </div>
                `;

        }

        function escapeHtml(str){ 
            return String(str).replace(/[&<>]/g, function(m){
                if(m==='&') return '&amp;'; 
                if(m==='<') return '&lt;'; 
                if(m==='>') return '&gt;'; 
                return m;
            }); 
        }
        
        function renderPagination() {
            const container = document.getElementById('paginationContainer');
            if (!container) return;
            
            let filtered = currentVideos.filter(v=> {
                if(currentCategory === 'all') {
                    return (!v.isUpcoming && !v.isTrending) && (!searchQuery || v.title.toLowerCase().includes(searchQuery.toLowerCase()));
                }
                return (v.category === currentCategory) && (!searchQuery || v.title.toLowerCase().includes(searchQuery.toLowerCase()));
            });
            
            totalPages = Math.ceil(filtered.length / itemsPerPage);
            
            if (totalPages <= 1) {
                container.innerHTML = '';
                return;
            }
            
            let html = '';
            
            let startPage = Math.max(1, currentPage - 2);
            let endPage = Math.min(totalPages, currentPage + 2);
            
            if (startPage <= 2) {
                startPage = 1;
                endPage = Math.min(totalPages, 5);
            }
            
            if (endPage >= totalPages - 1) {
                endPage = totalPages;
                startPage = Math.max(1, totalPages - 4);
            }
            
            if (startPage > 1) {
                html += `<button class="page-btn" onclick="goToPage(1)">1</button>`;
                if (startPage > 2) html += `<span style="color:#666; padding:0 5px;">...</span>`;
            }
            
            for (let i = startPage; i <= endPage; i++) {
                html += `<button class="page-btn ${currentPage === i ? 'active' : ''}" onclick="goToPage(${i})">${i}</button>`;
            }
            
            if (endPage < totalPages) {
                if (endPage < totalPages - 1) html += `<span style="color:#666; padding:0 5px;">...</span>`;
                html += `<button class="page-btn" onclick="goToPage(${totalPages})">${totalPages}</button>`;
            }
            
            html += `<span style="color:#888; margin-left: 15px; font-size: 14px;">Page ${currentPage} of ${totalPages}</span>`;
            
            container.innerHTML = html;
        }
        
        function goToPage(page) {

            currentPage = page;
            renderVideosWithPagination();
            renderPagination();
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }

            if(currentUser){ 
                let doc=await db.collection("users").doc(currentUser.uid).get(); 
                if(doc.exists){ 
                    currentUserMembership.active = doc.data().membershipActive || false; 
                    const badge = document.getElementById('membershipBadge');
                    if(badge) badge.style.display = currentUserMembership.active ? 'inline-block' : 'none';
                } 
            } 
        }
        
        auth.onAuthStateChanged(async(user)=>{ 
            if(user){ 
                let doc=await db.collection("users").doc(user.uid).get(); 
                if(doc.exists){ 
                    currentUser=user; 
                    currentUsername=doc.data().username;
                    currentUserMembership.active=doc.data().membershipActive||false; 
                    document.getElementById('authButtons').style.display='none'; 
                    document.getElementById('userInfo').style.display='flex'; 
                    document.getElementById('userName').innerHTML=doc.data().username; 
                    document.getElementById('membershipBadge').style.display=currentUserMembership.active?'inline-block':'none'; 
                } 
            } else { 
                currentUser=null; 
                currentUserMembership.active=false; 
                document.getElementById('authButtons').style.display='flex'; 
                document.getElementById('userInfo').style.display='none'; 
            } 
        });
        
        async function checkAndShowPopupNote() {
            try {
                const doc = await db.collection("popupNotes").doc("activeNote").get();
                if (doc.exists && doc.data().active === true) {
                    document.getElementById('popupNoteTitle').innerHTML = escapeHtml(doc.data().title);
                    document.getElementById('popupNoteMessage').innerHTML = escapeHtml(doc.data().message);
                    document.getElementById('popupNoteModal').classList.remove('hidden');
                }
            } catch (error) { console.error("Error:", error); }
        }
